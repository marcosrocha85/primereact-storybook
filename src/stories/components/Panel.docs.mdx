## Custom Header

<div style={{ width: "30rem" }}>
  <Panel
    header="Notifications"
    headerTemplate={(options) => {
      const {
        className,
        titleClassName,
        togglerClassName,
        togglerIconClassName,
        onTogglerClick,
      } = options;
      return (
        <div className={className}>
          <span className={titleClassName}>
            <i className="pi pi-bell mr-2"></i>
            Notifications
          </span>
          <div className="flex gap-2">
            <Button
              icon="pi pi-cog"
              text
              rounded
              severity="secondary"
              size="small"
              aria-label="Settings"
            />
            <button className={togglerClassName} onClick={onTogglerClick}>
              <span className={togglerIconClassName} />
            </button>
          </div>
        </div>
      );
    }}
    toggleable
  >
    <p className="m-0 line-height-3">
      Use <code>headerTemplate</code> to inject custom controls such as action
      buttons alongside the toggle affordance.
    </p>
  </Panel>
</div>
